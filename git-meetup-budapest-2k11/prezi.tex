\begin{frame}
\frametitle{Elosztott fejlesztés}
\begin{itemize}
\item Elosztottság (mint minden más modern verziókezelõ)
\item Távoli repók a helyi repókban branchként (fetch vs merge)
\item Repók közötti kommunikáció saját protokollal, HTTP-n, stb.
\item Többféle workflow támogatása
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Nemlineáris fejlesztés}
\begin{itemize}
\item Olcsó branch / merge
\item Eltérõ célok: merge framework
\item Eszközök nemlineáris projekttörténethez:
\begin{itemize}
\item Vizualizáció (gitk)
\item Navigálás (bisect)
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Nagy projektek hatékony kezelése}
\begin{itemize}
\item Nagyon gyors - nagy projektek hosszú történetét hatékonyan kezeli
\item Átlagban egy nagyságrenddel gyorsabb a többi verziókezelõnél (whygitisbetterthanx, kivétel: add)
\item Több nagyságrenddel gyorsabb néhány mûvelet esetén (branch/merge)
\item Extrém hatékony pack formátum: ver minden más szabadon elérhetõ megoldást
\item Implicit copy/rename detekció
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Projekttörténet kriptográfiai hitelesítése}
\begin{itemize}
\item Minden commit hash-e függ az összes hivatkozott objektum hash-étõl
\item A publikus objektumokat lehetetlen észrevétlenül módosítani
\item A tageket digitálisan alá lehet írni
\item Sebesség és biztonság egyszerre
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Eszközkészlet felépítés}
\begin{itemize}
\item Unix tradíció (1 eszköz 1 feladat)
\item Plumbing parancsok (API, C)
\item Porcelain parancsok (verziókezelõ, C/shell/perl)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Összefoglalás}
A Git egy elosztott verziókezelõ rendszer,
sebességre/hatékonyságra/használhatóságra optimalizálva. Az említett támogatott
tulajdonságok:
\begin{itemize}
\item Elosztott fejlesztés
\item Nemlineáris fejlesztés
\item Nagy projektek hatékony kezelése
\item Projekttörténet kriptográfiai hitelesítése
\item Eszközkészlet felépítés
\end{itemize}
\end{frame}
