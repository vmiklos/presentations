\begin{frame}
\frametitle{Google SoC}
\begin{itemize}
\item pályázati lehetõség
\item szervezet pályázhatnak, projektötletekkel
\item tanulók pályázhatnak, az ötletek alapján saját kidolgozott projektekkel
\item szabad szoftverek fejlesztése, de a tanuló pénzt kapnak a Google-tõl
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{OpenOffice.org fejlesztés}
\begin{itemize}
\item kihívás: nagy projekt (split build még nem upstream)
\item 8 millió kódsor (leginkább C++)
\item fordítás trükközve is legalább 4-5 óra
\item forráskód mérete kb. 4 GB
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{OpenOffice.org fejlesztés}
\framesubtitle{LibreOffice vs. OpenOffice.org}
\begin{itemize}
\item Novell/Redhat/Google/IBM/etc. által szponzorált patchset / fork
\item 15+ fejlesztõ
\item Szabadabb license-elés (fejleszõknek nem kell copyright-ot átadni)
\item Szabadabb politika (felhasználóknak nincs regisztráció)
\item Gyakorlatilag az összes Linux disztribúció ezt használja
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{OpenOffice.org fejlesztés}
\framesubtitle{LibreOffice vs. OpenOffice.org: fejlesztõi szemmel}
Fordítás:
\begin{itemize}
\item OpenOffice.org
\begin{verbatim}
./configure --with-use-shell=bash \
	--disable-build-mozilla \
	--with-jdk-home=/usr/lib/jvm/java-1.5.0-gcj \
	--with-system-mozilla=mozilla \
	--with-openldap \
	--disable-binfilter --disable-epm
make
export LOCALINSTALLDIR=~/git/gsoc/\
	upstream/myhack-install
cd ~/git/gsoc/\
	upstream/myhack/instsetoo_native/util
rm -rf ../../../myhack-install
dmake openoffice_en-US PKGFORMAT=installed
\end{verbatim}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{OpenOffice.org fejlesztés}
\framesubtitle{LibreOffice vs. OpenOffice.org: fejlesztõi szemmel}
Fordítás:
\begin{itemize}
\item LibreOffice
\begin{verbatim}
./configure --with-distro=Frugalware
make
make dev-install
\end{verbatim}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{RTF export fejlesztés}
\begin{itemize}
\item ötlet: az RTF sokban hasonlít a doc/docx-re (Microsoft találmány)
\item a doc / docx exporternek már van egy közös alapja
\item készítsünk egy új exportert erre a közös alapra építve
\item cél: tudjon mindent amit a régi exporter, kisebb méret, új funkciók
\item integráció: Go-OO 3.3 / OpenOffice.org 3.4 (várhatóan)
\end{itemize}
\end{frame}
